from typing import Any


class Singylist:
	def __init__(self):
		self.head = self
		self.data: Any = None
		self.new_pt = None
		self.end = None

	def append(self, data):
		if self.new_pt is not None:
			self.end.data = data
			self.end.new_pt = Singylist()
			self.end = self.end.new_pt
		else:
			self.data = data
			self.new_pt = Singylist()
			self.end = self.new_pt

	def __str__(self):
		res = []
		tmp = self.head
		while tmp.new_pt is not None:
			res.append(tmp.data)
			tmp = tmp.new_pt
		return str(res)


if __name__ == '__main__':
	A = Singylist()
	A.append(1)
	A.append(2)
	A.append(3)
	A.append(4)

	print(A)
